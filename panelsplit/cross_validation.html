 27def _nunique_subset(data, indices):
 28    """
 29    Helper function to get number of unique values in a subset of data.
 30    Works with both pandas and narwhals-compatible data.
 31    """
 32    data_nw = nw.from_native(data, pass_through=True)
 33    if hasattr(data_nw, "filter") and hasattr(data_nw, "n_unique"):
 34        return data_nw.filter(indices).n_unique()
 35    else:
 36        # Fallback to numpy operations
 37        subset_data = (
 38            data[indices] if hasattr(data, "__getitem__") else data.loc[indices]
 39        )
 40        return len(np.unique(subset_data))

131    def _split_unique_periods(self, indices, unique_periods):
132        """
133        Split unique periods into training and testing sets based on TimeSeriesSplit indices.
134
135        Parameters
136        ----------
137        indices : iterator
138            An iterator yielding pairs of train and test indices from TimeSeriesSplit.
139        unique_periods : array-like
140            The collection of unique periods.
141
142        Returns
143        -------
144        list of tuple of np.ndarray
145            A list where each tuple contains two numpy arrays:
146            (unique_train_periods, unique_test_periods) for each split.
147        """
148        unique_periods_array = _to_numpy_array(unique_periods)
149        u_periods_cv = []
150        for i, (train_index, test_index) in enumerate(indices):
151            unique_train_periods = unique_periods_array[train_index]
152            unique_test_periods = unique_periods_array[test_index]
153            if (i == 0) & self._include_first_train_in_test:
154                unique_test_periods = np.concatenate(
155                    [unique_train_periods, unique_test_periods]
156                )
157            elif (i > 0) & self._include_train_in_test:
158                unique_test_periods = np.concatenate(
159                    [unique_train_periods, unique_test_periods]
160                )
161            u_periods_cv.append((unique_train_periods, unique_test_periods))
162        return u_periods_cv

164    def _gen_splits(self) -> List[Tuple[np.ndarray, np.ndarray]]:
165        """
166        Generate integer index arrays for training and testing sets for each split.
167
168        Returns
169        -------
170        List[Tuple[np.ndarray, np.ndarray]]
171            A list where each tuple contains two integer arrays:
172            (train_indices, test_indices) corresponding to each split.
173        """
174        train_test_splits = []
175
176        # Convert snapshots once if they exist
177        snapshots_array = (
178            _to_numpy_array(self._snapshots) if self._snapshots is not None else None
179        )
180
181        for i, (train_periods, test_periods) in enumerate(self._u_periods_cv):
182            if self._snapshots is not None:
183                if test_periods.max() >= self._snapshots.min():
184                    snapshot_val = test_periods.max()
185                else:
186                    snapshot_val = self._snapshots.min()
187                    warnings.warn(
188                        (
189                            f"The maximum period value {test_periods.max()} for split {i} is less than "
190                            f"the minimum snapshot value {self._snapshots.min()}. Defaulting to minimum "
191                            f"snapshot value for split {i}."
192                        ),
193                        stacklevel=2,
194                    )
195                # Create boolean masks then convert to integer indices
196                train_period_mask = np.isin(self._periods, train_periods)
197                test_period_mask = np.isin(self._periods, test_periods)
198
199                # Handle snapshots
200                snapshot_mask = snapshots_array == snapshot_val
201
202                train_indices = np.where(train_period_mask & snapshot_mask)[0]
203                test_indices = np.where(test_period_mask & snapshot_mask)[0]
204            else:
205                # Create boolean masks then convert to integer indices
206                train_indices = np.where(np.isin(self._periods, train_periods))[0]
207                test_indices = np.where(np.isin(self._periods, test_periods))[0]
208
209            train_test_splits.append((train_indices, test_indices))
210
211        return train_test_splits

376    def gen_snapshots(
377        self, data: IntoDataFrame, period_col: Optional[str] = None
378    ) -> IntoDataFrame:
379        """
380        Generate snapshots for each cross-validation split.
381
382        Parameters
383        ----------
384        data : IntoDataFrame
385            The DataFrame from which snapshots are generated.
386        period_col : str, optional
387            The column name (or index) in `data` to derive the snapshot period. If both an index
388            and a column in `data` are named `period_col`, the index is used by default.
389
390        Returns
391        -------
392        IntoDataFrame
393            A concatenated DataFrame containing snapshots for each split. Each snapshot includes
394            a 'split' column (indicating the split number) and, if `period_col` is provided, a
395            'snapshot_period' column.
396
397        Examples
398        --------
399        >>> import pandas as pd
400        >>> data = pd.DataFrame({
401        ... 'value': [10, pd.NA, 30],
402        ... 'period': [1, 2, 3],
403        ... }).astype('Int32')
404        >>> ps = PanelSplit(periods=data['period'], n_splits=2)
405        >>> snapshots = ps.gen_snapshots(data, period_col='period')
406        >>> print(snapshots)
407           value  period  split  snapshot_period
408        0     10       1      0                2
409        1   <NA>       2      0                2
410        0     10       1      1                3
411        1   <NA>       2      1                3
412        2     30       3      1                3
413        """
414        periods = get_index_or_col_from_df(data, period_col)
415        periods = check_periods(periods)
416
417        # Use narwhals for dataframe-agnostic operations
418        data_nw = nw.from_native(data, pass_through=True)
419        periods_nw = nw.from_native(periods, pass_through=True)
420
421        splits = self.split()
422        snapshots = []
423        for i, split in enumerate(splits):
424            # Combine train and test indices (both are already integer arrays)
425            row_indices = np.unique(np.concatenate([split[0], split[1]]))
426
427            split_data = _safe_indexing(data_nw, row_indices)
428
429            if period_col is not None:
430                # Get periods for this split and find max
431                split_periods = _safe_indexing(periods_nw, row_indices, to_native=True)
432                last_period = np.max(np.unique(split_periods))
433
434                # Add columns using narwhals
435                split_data = split_data.with_columns(
436                    [
437                        nw.lit(i).alias("split"),
438                        nw.lit(last_period).alias("snapshot_period"),
439                    ]
440                )
441            else:
442                split_data = split_data.with_columns([nw.lit(i).alias("split")])
443
444            snapshots.append(split_data)
445
446        return _safe_indexing(nw.concat(snapshots), to_native=True)

449def drop_splits(cv, y):
450    """
451    Drop cross-validation splits if either the training or testing set is empty or contains only one unique value.
452
453    Parameters
454    ----------
455    cv : list
456        A list of tuples, where each tuple contains (train_indices, test_indices) as boolean arrays.
457        The object is expected to have an attribute `n_splits` (an integer) and support the `pop` method.
458    y : IntoSeries
459        Series of shape (n_samples,) containing target values.
460
461    Returns
462    -------
463    list
464        The modified list of splits with the problematic splits removed.
465
466    Examples
467    --------
468    >>> import pandas as pd
469    >>> data = pd.DataFrame({
470    ... 'value': [10, pd.NA, 30],
471    ... 'period': [1, 2, 3],
472    ... }).astype('Int32')
473    >>> ps = PanelSplit(periods=periods, n_splits=2)
474    >>> drop_splits(ps, data['value'])
475    Dropping split 0 as either the test or train set is either empty or contains only one unique value.
476    """
477    for i, (train_indices, test_indices) in enumerate(cv.split()):
478        if (len(train_indices) == 0 or len(test_indices) == 0) or (
479            _nunique_subset(y, train_indices) == 1
480            or _nunique_subset(y, test_indices) == 1
481        ):
482            cv.n_splits -= 1
483            cv.train_test_splits.pop(i)
484            print(
485                f"Dropping split {i} as either the test or train set is either empty or contains only one unique value."
486            )
487    return cv

377    def gen_snapshots(
378        self, data: IntoDataFrame, period_col: Optional[str] = None
379    ) -> IntoDataFrame:
380        """
381        Generate snapshots for each cross-validation split.
382
383        Parameters
384        ----------
385        data : IntoDataFrame
386            The DataFrame from which snapshots are generated.
387        period_col : str, optional
388            The column name (or index) in `data` to derive the snapshot period. If both an index
389            and a column in `data` are named `period_col`, the index is used by default.
390
391        Returns
392        -------
393        IntoDataFrame
394            A concatenated DataFrame containing snapshots for each split. Each snapshot includes
395            a 'split' column (indicating the split number) and, if `period_col` is provided, a
396            'snapshot_period' column.
397
398        Examples
399        --------
400        >>> import pandas as pd
401        >>> data = pd.DataFrame({
402        ... 'value': [10, pd.NA, 30],
403        ... 'period': [1, 2, 3],
404        ... }).astype('Int32')
405        >>> ps = PanelSplit(periods=data['period'], n_splits=2)
406        >>> snapshots = ps.gen_snapshots(data, period_col='period')
407        >>> print(snapshots)
408           value  period  split  snapshot_period
409        0     10       1      0                2
410        1   <NA>       2      0                2
411        0     10       1      1                3
412        1   <NA>       2      1                3
413        2     30       3      1                3
414        """
415        periods = get_index_or_col_from_df(data, period_col)
416        periods = check_periods(periods)
417
418        # Use narwhals for dataframe-agnostic operations
419        data_nw = nw.from_native(data, pass_through=True)
420        periods_nw = nw.from_native(periods, pass_through=True)
421
422        splits = self.split()
423        snapshots = []
424        for i, split in enumerate(splits):
425            # Combine train and test indices (both are already integer arrays)
426            row_indices = np.unique(np.concatenate([split[0], split[1]]))
427
428            split_data = _safe_indexing(data_nw, row_indices)
429
430            if period_col is not None:
431                # Get periods for this split and find max
432                split_periods = _safe_indexing(periods_nw, row_indices, to_native=True)
433                last_period = np.max(np.unique(split_periods))
434
435                # Add columns using narwhals
436                split_data = split_data.with_columns(
437                    [
438                        nw.lit(i).alias("split"),
439                        nw.lit(last_period).alias("snapshot_period"),
440                    ]
441                )
442            else:
443                split_data = split_data.with_columns([nw.lit(i).alias("split")])
444
445            snapshots.append(split_data)
446
447        return _safe_indexing(nw.concat(snapshots), to_native=True)
